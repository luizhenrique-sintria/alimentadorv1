async function conectar() { try { console.log("Buscando dispositivo..."); const device = await navigator.bluetooth.requestDevice({ // Filtra pelo UUID do serviço em vez de filtrar por nome filters: [{ services: [SERVICE_UUID] }] }); console.log("Conectando ao servidor GATT..."); const server = await device.gatt.connect(); console.log("Obtendo serviço..."); const service = await server.getPrimaryService(SERVICE_UUID); // Configurar Recebimento (Peso) const txChar = await service.getCharacteristic(TX_UUID); await txChar.startNotifications(); txChar.addEventListener('characteristicvaluechanged', e => { const valor = new TextDecoder().decode(e.target.value); document.getElementById('peso').innerText = valor; }); // Configurar Envio (Tara) characteristicRX = await service.getCharacteristic(RX_UUID); alert("Balança Conectada!"); } catch (err) { console.error("Erro detalhado:", err); alert("Erro: " + err.message); } }
